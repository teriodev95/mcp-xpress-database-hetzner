\section{Metodolog\'ia}

\subsection{Fuentes de Datos}

\begin{enumerate}[leftmargin=2em]
  \item \textbf{vw\_cobranza\_snapshots\_reportes\_generales}: Vista que cruza
    \texttt{cobranza\_snapshots} con \texttt{debitos\_historial}. Columnas clave:
    \texttt{sucursal}, \texttt{gerencia}, \texttt{agencia}, \texttt{rendimiento\_miercoles},
    \texttt{adelanto\_miercoles}. Filtro: \texttt{dia\_semana\_es = 'Miercoles'}, \texttt{hora = 20}.
  \item \textbf{pagos\_v3}: Pagos individuales. Filtro por \texttt{Created\_at} del mi\'ercoles
    de cada semana.
  \item \textbf{prestamos\_v2}: Pr\'estamos activos. Campo \texttt{Dia\_de\_pago} para verificar
    que el cliente tiene asignado JUEVES.
\end{enumerate}

\subsection{Criterios de Detecci\'on}

\begin{itemize}[leftmargin=2em]
  \item \texttt{rendimiento\_miercoles} $> 100\%$ en la vista.
  \item \texttt{adelanto\_miercoles} $> 0$ (monto que excede el d\'ebito del mi\'ercoles).
  \item Confirmaci\'on con \texttt{pagos\_v3}: existen pagos registrados el mi\'ercoles para
    clientes cuyo \texttt{Dia\_de\_pago = 'JUEVES'} en \texttt{prestamos\_v2}.
  \item \textbf{Consistente}: el patr\'on se repite en ambas semanas.
\end{itemize}

\subsection{F\'ormulas de la Vista}

\begin{itemize}[leftmargin=2em]
  \item \texttt{rendimiento\_miercoles} = \texttt{cobranza\_pura} / \texttt{debito\_miercoles} $\times$ 100
  \item \texttt{adelanto\_miercoles} = MAX(\texttt{cobranza\_pura} $-$ \texttt{debito\_miercoles}, 0)
  \item \texttt{rendimiento\_jueves} = \texttt{cobranza\_pura} / (\texttt{debito\_mie} + \texttt{debito\_jue}) $\times$ 100
  \item \texttt{adelanto\_jueves} = MAX(\texttt{cobranza\_pura} $-$ (\texttt{debito\_mie} + \texttt{debito\_jue}), 0)
\end{itemize}
